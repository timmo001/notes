import { Card, CardGrid, LinkCard } from "@astrojs/starlight/components";

`notes` manages repository notes in a Git-backed vault and exposes the same behaviour through a CLI and an MCP server.

<CardGrid>
  <Card title="Project notes" icon="open-book">
    Store Markdown notes under `projects/{owner}/{repo}` using a Git remote or a
    local project identity.
  </Card>
  <Card title="Safe writes" icon="setting">
    Reads, writes, and deletes are constrained to the notes vault, then
    committed and best-effort pushed.
  </Card>
  <Card title="Handoffs" icon="document">
    Handoffs are ordinary notes tagged `handoff`, using the same priority
    metadata as other notes.
  </Card>
  <Card title="MCP server" icon="server">
    The same note operations are exposed as raw-text MCP tools for agent
    harnesses.
  </Card>
</CardGrid>
